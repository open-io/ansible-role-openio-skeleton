cmds=("ansible-lint ." "yamllint -c .yamllint .")
for cmd in "${cmds[@]}"; do
  echo "Running ${cmd%% *}"
  cmd_out="$($cmd)"
  echo -n "${cmd_out}"
  if [ "$cmd_out" ]; then
      echo -e "\nRejecting commit: ${cmd%% *} returned errors"
      exit 1
  fi
done
EOF
